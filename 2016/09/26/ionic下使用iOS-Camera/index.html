在创建好 ionic 项目,并添加了基本的 cordova 插件后

安装 cordova 相机插件和 ngCordova

要使用 iOS Camera 需要添加 cordova 相机插件和 ngCordova 这个类库,
安装 cordova 相机插件
cordova plugin add org.apache.cordova.camera
安装 ngCordova
安装这个类库可以使用 bower 工具, bower 安装
npm install -g bower
注意,切换到 ionic 项目根目录下,安装 ngCordova
bower install ngCordova

ionic 下 iOS Camera 的调用

.controller(‘AddressEditCtrl’, function($scope, $cordovaCamera) {

$scope.takephotoimage = “img/address/address_person.jpg”;

$scope.takePicture = function() {

var options = {
  quality : 100, //拍照保存的照片清晰度,0~100,100最佳
  destinationType : Camera.DestinationType.DATA_URL, //返回值的形式,这里是用的是base64编码格式
  sourceType : Camera.PictureSourceType.CAMERA, //设置照片的资源
  allowEdit : true, //选择照片钱是否可编辑
  encodingType: Camera.EncodingType.JPEG, //返回的图标的是个,这里是jpeg格式
  targetWidth: 50, //照片像素
  targetHeight: 50,
  mediaType: 0, //媒体类型
  popoverOptions: CameraPopoverOptions,
  saveToPhotoAlbum: false //是否保存照片到本地
};

$cordovaCamera.getPicture(options).then(function(imageData) {
  $scope.imgURI = "data:image/jpeg;base64," + imageData;
  console.log("拍照 1 "+$scope.imgURI);
  $scope.takephotoimage = "data:image/jpeg;base64," + imageData;
}, function(err) {
  // An error occured. Show a message to the user
  console.log("拍照 2 "+err);
});
